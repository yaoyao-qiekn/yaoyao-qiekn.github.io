Fancybox

thanks to cungudafa in csdn

in head.ejs import fancybox.min.css and remove zoom.css

<link rel="stylesheet" href="/css/jquery.fancybox.min.css" media="all">
<!-- 删去zoom 2022-10-09 22:38:12
<link rel="stylesheet" href="/css/zoom.css" media="all">
-->

in footer.ejs import fancy.min.js and remove zoom.css.min.js

<script type="text/javascript" src="/js/jquery.fancybox.min.js"></script>
<!-- 删去zoom 2022-10-09 22:38:43
<script type="text/javascript" src="/js/zoom.min.js"></script>
-->

then add in footer.ejs

<!-- fancybox -->
<script>
  $('[data-fancybox="images"]').fancybox({//可选
    thumbs : {
      autoStart : true //缩略图
    }
  });
  $('[data-fancybox]').fancybox({//启用函数，必须
    protect: true //图片右键不能下载，可选
  });
</script>

finally

modify Renderer.prototype.image function in node_modules\marked\lib\marked.js

 //original
 var out = '<img src="' + href + '" alt="' + text + '"';
 //changed
 var out = '<a data-fancybox="gallery" data-caption="'+text+'" href="'+href+'"><img data-action="zoom" src="' + href + '" alt="' + text + '"';

Code-block-fullscreen

At first use javascript to add class “highlight-warp” in <pre>, looks like

<pre class="hightlight-wrap">
    <code>...</code>
</pre>

Pay attention that this javascript below is out of date. For example :eq() is deprecated in jQuery docs. The original author is Mashrio , and I haven’t learned javascript yet. I try this javascript below, buged in line 23. I do hljs.highlightAll(); in chrome console and add the class="highlight-wrap manually, it works(GIF below). So search the web to find solution.

function code_highlight_style () {
  function gen_top_bar (i) {
    var attributes = {
      'autocomplete': 'off',
      'autocorrect': 'off',
      'autocapitalize': 'off',
      'spellcheck': 'false',
      'contenteditable': 'false',
      'design': 'by hojun'
    }
    var ele_name = $('pre:eq(' + i + ')')[0].children[0].className
    var lang = ele_name.substr(0, ele_name.indexOf(' ')).replace('language-', '')
    if (lang.toLowerCase() == 'hljs') var lang = 'text'
    if (lang.toLowerCase() == 'js') var lang = 'javascript'
    if (lang.toLowerCase() == 'md') var lang = 'markdown'
    if (lang.toLowerCase() == 'py') var lang = 'python'
    $('pre:eq(' + i + ')').addClass('highlight-wrap')
    for (var t in attributes) {
      $('pre:eq(' + i + ')').attr(t, attributes[t])
    }
    $('pre:eq(' + i + ') code').attr('data-rel', lang.toUpperCase())
  }
  $('pre code').each(function (i, block) {
    hljs.highlightBlock(block)
  })
  for (var i = 0; i < $('article pre').length; i++) {
    gen_top_bar(i)
  }
  $('pre').on('click', function (e) {
    if (e.target !== this) return
    $(this).toggleClass('code-block-fullscreen')
    $('html').toggleClass('code-block-fullscreen-html-scroll')
  })
  hljs.initLineNumbersOnLoad()
}
try {
  code_highlight_style()
} catch (e) {}

/* code-block-fullscreen */
.code-block-fullscreen {
    position:fixed;
    top:0;
    left:0;
    width:100%;
    height:100%;
    min-width:100%;
    z-index:9999999;
    margin:0;
    animation:elastic 1s
}
.code-block-fullscreen code {
    --widthA:100%;
    --widthB:calc(var(--widthA) - 30px);
    height:var(--widthB);
    min-height:99%;
    overflow-y:hidden;
    overflow-x:auto;
    height:auto
}
.code-block-fullscreen-html-scroll {
    overflow:hidden
}

    0%   { transform:scale(0) }
    55%  { transform:scale(1) }

    100% { transform:scale(1) }
